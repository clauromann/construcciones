\subsection{Mediciones}
\label{subsec:mediciones}
% -----------------------------------------------------------

Las mediciones se obtienen directamente de la geometría adoptada
(sección~\ref{sec:muro_predim}) y de los cálculos de materiales
del proceso constructivo (sección~\ref{sec:muro_proceso}):

\begin{align*}
  V_{\mathrm{exc}}    &= (B + 2 \times 0{,}60) \times (h_z + e_{\mathrm{HM}} + D_f)
                          \times L
                       = 5{,}60 \times 1{,}90 \times 150
                       = \mathbf{1\,596~\text{m}^3} \\
  V_{\mathrm{HM-15}}  &= B \times e_{\mathrm{HM}} \times L
                       = 4{,}40 \times 0{,}10 \times 150
                       = \mathbf{66~\text{m}^3} \\
  V_{\mathrm{HA-30}}  &= \underbrace{4{,}40 \times 0{,}80}_{V_{\mathrm{zapata}}}
                         \times 150
                       + \underbrace{\tfrac{0{,}60+0{,}75}{2} \times 6{,}50}
                                    _{V_{\mathrm{fuste}}} \times 150
                       = 528 + 659 = \mathbf{1\,187~\text{m}^3} \\
  P_{\mathrm{acero}}  &= 288{,}3~\text{kg/ml} \times 150~\text{ml}
                       = \mathbf{43\,245~\text{kg}} \\
  S_{\mathrm{enc,fuste}} &= 2 \times 6{,}50 \times 150
                          = \mathbf{1\,950~\text{m}^2} \\
  S_{\mathrm{enc,zap}}   &= 2 \times 0{,}80 \times 150
                          = \mathbf{240~\text{m}^2} \\
  S_{\mathrm{imp}}    &= 6{,}50 \times 150 = \mathbf{975~\text{m}^2} \\
  V_{\mathrm{rell}}   &= B_{\mathrm{talón}} \times H_f \times L
                       = 2{,}85 \times 6{,}50 \times 150
                       = \mathbf{2\,779~\text{m}^3}
\end{align*}

\begin{notabox}[Coste unitario por ml de muro]
  PEM por metro lineal de muro:
  $371\,094 \div 150 = \mathbf{2\,474~\text{\euro/ml}}$.
  La partida más significativa es el \textbf{encofrado del fuste} (34\,\% del PEM),
  seguida del \textbf{hormigón HA-30} (34\,\%) y el
  \textbf{acero} (16\,\%).
  Este desglose es coherente con los ratios habituales para muros de contención de
  hormigón armado de altura $H > 5$~m.
\end{notabox}
